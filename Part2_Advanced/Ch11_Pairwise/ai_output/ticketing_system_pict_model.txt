# =========================
# 購票系統 PICT Model
# =========================

# 參數定義（Parameters）
票種: 成人票, 學生票, 敬老票, 早鳥優惠票
座位: VIP區, 搖滾區, 看台區
付款方式: 信用卡, LINEPay, ATM轉帳
優惠: 無, 折扣券, 好友推薦碼
取票方式: 電子票券, 超商取票, 實體郵寄

# =========================
# 限制條件（Constraints）
# =========================

# 限制 1：
# 學生票 → 只能超商取票
IF [票種] = "學生票" THEN [取票方式] = "超商取票";

# 限制 2：
# 早鳥優惠票 → 不可使用任何優惠
IF [票種] = "早鳥優惠票" THEN [優惠] = "無";

# 限制 3：
# ATM 轉帳 → 不可實體郵寄
IF [付款方式] = "ATM轉帳" THEN [取票方式] <> "實體郵寄";
